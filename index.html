핸드폰 포커스 체크 연동

Supabase에 아래 테이블을 한 번 만든 뒤 URL과 anon key를 저장하면, PC 대시보드와 핸드폰 포커스 화면이 같은 상태를 봅니다. Public Dashboard URL을 입력하면 모바일 링크가 file:///가 아니라 배포 도메인 기준으로 복사됩니다. 
create table if not exists public.nova_sync_state (
  id text primary key,
  state jsonb not null,
  updated_at timestamptz default now(),
  client_id text
);

alter table public.nova_sync_state enable row level security;

drop policy if exists "nova sync read" on public.nova_sync_state;
drop policy if exists "nova sync write" on public.nova_sync_state;
drop policy if exists "nova sync update" on public.nova_sync_state;

create policy "nova sync read" on public.nova_sync_state
for select using (true);

create policy "nova sync write" on public.nova_sync_state
for insert with check (true);

create policy "nova sync update" on public.nova_sync_state
for update using (true) with check (true);
